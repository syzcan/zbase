@Controller
@RequestMapping(value = "/demo")
public class DemoController extends BaseController {
	private final static Logger logger = Logger.getLogger(DemoController.class);
	@Autowired
	private DemoService demoService;

	/**
	 * 查询demo详情data
	 */
	@ResponseBody
	@RequestMapping(value = "/data")
	public PageData data(Demo demo) {
		PageData pd = new PageData("errMsg", "success");
		demo = demoService.loadDemo(demo);
		pd.put("data", demo);
		return pd;
	}
}

this.formJson = function($form) {
	var array = $form.serializeArray();
	var json = {};
	$.each(array,
	function(i, field) {
		if (json[field.name]) {
			if ($.isArray(json[field.name])) {
				json[field.name].push(field.value);
			} else {
				json[field.name] = [json[field.name], field.value];
			}
		} else {
			json[field.name] = field.value;
		}
	});
	return json;
}
